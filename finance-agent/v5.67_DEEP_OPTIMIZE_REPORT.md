```python
def strategy_performance_weighting(ranked: list, backtest_metrics: dict = None) -> list:
    """v5.67: 基于回测Sharpe/MaxDD动态调整赛道权重
    
    融合: MACD+RSI(17.1%, 2.35), 新能源(14.66%, 1.78), 消费(Sharpe 0.85)
    
    Args:
        ranked: score_and_rank()输出
        backtest_metrics: 从backtester获取的性能指标
            {
                'MACD_RSI': {'return': 0.171, 'sharpe': 2.35, 'maxdd': 0.0408},
                'MULTI_FACTOR': {...},
                'NEW_ENERGY': {...}
            }
    
    Returns: 应用权重后的排序list
    """
    try:
        from config import STRATEGY_PERFORMANCE_WEIGHT
        
        if not ranked or not backtest_metrics:
            return ranked
        
        for stock in ranked:
            signals = stock.get('signals', [])
            score = stock.get('score', 0)
            
            # 判定策略类型
            strategy_type = 'unknown'
            if any('MACD' in str(s) or 'RSI' in str(s) for s in signals):
                strategy_type = 'MACD_RSI'
            elif any('多因子' in str(s) for s in signals):
                strategy_type = 'MULTI_FACTOR'
            
            # 查询该策略的Sharpe
            metrics = backtest_metrics.get(strategy_type, {})
            sharpe = metrics.get('sharpe', 1.0)
            maxdd = metrics.get('maxdd', 0.05)
            
            # 应用权重: Sharpe倍数 × MaxDD惩罚
            weight_multiplier = 1.0
            
            if sharpe >= 2.0:
                weight_multiplier *= STRATEGY_PERFORMANCE_WEIGHT['sharpe_good']['weight_multiplier']
            elif sharpe >= 1.5:
                weight_multiplier *= STRATEGY_PERFORMANCE_WEIGHT['sharpe_normal']['weight_multiplier']
            elif sharpe < 1.0:
                weight_multiplier *= STRATEGY_PERFORMANCE_WEIGHT['sharpe_weak']['weight_multiplier']
            
            # MaxDD惩罚
            if maxdd > 0.05:
                weight_multiplier *= STRATEGY_PERFORMANCE_WEIGHT['maxdd_penalty']['penalty_multiplier']
            
            stock['score'] = int(stock['score'] * weight_multiplier)
            stock['_strategy_weight_applied'] = f"{weight_multiplier:.2f}x"
        
        ranked.sort(key=lambda x: -x.get('score', 0))
        return ranked
    except Exception as e:
        print(f"⚠️ strategy_performance_weighting失败: {e}")
        return ranked
```

```python
def sector_concentration_limit(ranked: list, positions: list) -> list:
    """v5.67: 赛道集中度限制
    
    防止单赛道崩塌风险:
    - 科技成长: 最多8只
    - 新能源: 最多6只
    - 其他赛道: 最多3只
    """
    try:
        from performance_tracker import classify_sector
        
        if not ranked:
            return ranked
        
        # 统计当前持仓赛道分布
        sector_count = {}
        for pos in positions:
            code = pos.get('symbol', '')
            name = pos.get('name', '')
            if not code:
                continue
            sector = classify_sector(code, name)
            sector_count[sector] = sector_count.get(sector, 0) + 1
        
        # 赛道集中度限制
        sector_limits = {
            '科技成长': 8,
            '新能源': 6,
            '其他': 3
        }
        
        filtered = []
        for stock in ranked:
            code = stock.get('code', '')
            name = stock.get('name', '')
            sector = classify_sector(code, name)
            
            limit = sector_limits.get(sector, 3)
            current_count = sector_count.get(sector, 0)
            
            if current_count >= limit:
                stock['_excluded_reason'] = f"赛道{sector}已达集中度上限({limit})"
                continue
            
            filtered.append(stock)
            sector_count[sector] = current_count + 1
        
        return filtered
    except Exception as e:
        print(f"⚠️ sector_concentration_limit失败: {e}")
        return ranked
```

```python
def check_dynamic_stop_v2(positions: list, quotes: dict, regime: str = "") -> list:
    """v5.67: 追踪止损v2 - 基于MaxDD自动调整止损线
    
    目标: MaxDD 4.08% → 3.2% (-20%)
    
    规则:
    - 高风险(Sharpe<1.0): 回撤4.0%快速止损
    - 中风险(Sharpe 1.0-1.5): 回撤4.5%止损
    - 低风险(Sharpe>1.5): 回撤5.0%止损
    - 时间止损: 15天无新高自动止损(避免僵尸持仓)
    """
    results = []
    for pos in positions:
        symbol = pos['symbol']
        quote = quotes.get(symbol, {})
        current_price = quote.get('price', pos.get('current_price', 0))
        if current_price <= 0:
            continue
        
        peak_price = pos.get('peak_price', pos.get('buy_price', current_price))
        profit_pct = (current_price - peak_price) / peak_price if peak_price else 0
        
        # 获取该持仓的Sharpe (从entry_quality_score中推断)
        sharpe_level = pos.get('_sharpe_level', 'medium')  # default: medium risk
        
        # 根据Sharpe等级选择回撤阈值
        if sharpe_level == 'high':  # Sharpe > 1.5
            aggressive_retracement = 0.05
        elif sharpe_level == 'medium':  # Sharpe 1.0-1.5
            aggressive_retracement = 0.045
        else:  # Sharpe < 1.0
            aggressive_retracement = 0.04
        
        # 追踪止损检查
        if profit_pct < -aggressive_retracement:
            results.append({
                'symbol': symbol,
                'action': 'sell',
                'reason': f'追踪止损: 回撤{profit_pct:.2%} < -{aggressive_retracement:.1%}',
                'price': current_price
            })
        
        # 时间止损: 15天无新高
        from position_manager import _trading_days_since
        hold_days = _trading_days_since(pos.get('buy_date', ''))
        if hold_days >= 15:
            days_since_peak = pos.get('_days_since_peak', 0)
            if days_since_peak >= 15 and profit_pct < 0.05:  # 浮盈<5%且无新高
                results.append({
                    'symbol': symbol,
                    'action': 'sell',
                    'reason': f'时间止损: 持仓{hold_days}天无新高',
                    'price': current_price
                })
    
    return results
```
