## 🎓 使用示例

### 示例1: 检查策略匹配

```python
from v5_77_strategy_fusion import check_optimal_strategy_match

# 候选股票数据
candidate = {
    'code': '000001',
    'name': '平安银行',
    'tech_indicators': {
        'macd_signal': 'golden_cross',      # MACD买入信号
        'macd': 0.5,
        'macd_signal_line': 0.2,
        'rsi14': 28,                        # RSI超卖
        'macd_days_since_cross': 1,         # 最近1天金叉
        'institution_holding_pct': 0.18,    # 机构持股18%
    }
}

# 检查匹配
matched, info = check_optimal_strategy_match(
    candidate['code'],
    candidate['name'],
    candidate['tech_indicators'],
    signals=['MACD_GC', 'RSI_OVERSOLD'],
    sector='金融'
)

# 输出
if matched:
    print(f"✅ 命中最优策略! 置信度: {info['confidence']}")
    # 输出: ✅ 命中最优策略! 置信度: 85
```

### 示例2: 准确率分析

```python
from v5_77_accuracy_tracker import AccuracyTracker

tracker = AccuracyTracker()

# 分析最近30天准确率
analysis = tracker.analyze_accuracy_period(days_back=30)

print(f"样本数: {analysis['sample_size']}")
print(f"命中率: {analysis['hit_rate_pct']:.1f}%")
print(f"平均收益: {analysis['avg_return_pct']:.2f}%")
print(f"Sharpe比: {analysis['sharpe_ratio']:.2f}")

# 按赛道分解
for sector, stats in analysis['sector_breakdown'].items():
    print(f"  {sector}: {stats}")

tracker.close()
```
